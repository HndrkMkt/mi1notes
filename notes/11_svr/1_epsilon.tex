\begin{frame}\frametitle{Regression setting}

\mode<article>{
\underline{The Setting}:
}

\begin{itemize}
\item[] \underline{Data}:\\
    \begin{equation}
    \big\{ (\vec x^{(\alpha)}, y_{T}^{(\alpha)}\big\}_{\alpha=1}^{p}
    \end{equation}
    with $\vec x \in \R^{N}$ and $y_{T} \in \R${}

\item[] \underline{Model}:\\
    linear neuron:
    \begin{equation}
        y(\vec x; \vec w, b) = \vec w^{\top} \vec x + b
    \end{equation}


    
\end{itemize}
